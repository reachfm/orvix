## Deployment Package Build Commands

```bash
# Step 1: Build frontend apps
cd web
npm install
npm run build
cd ..

# Step 2: Build backend binary
go build -ldflags="-s -w \
  -X main.Version=0.1.0 \
  -X main.Product=OrvixEM \
  -X main.Commit=$(git rev-parse --short HEAD) \
  -X main.Channel=stable \
  -X main.BuildDate=$(date -u +%Y-%m-%dT%H:%M:%SZ)" \
  -o dist/orvix-linux-amd64-v0.1.0 ./cmd/orvix

# Step 3: Generate checksums
cd dist
sha256sum orvix-linux-amd64-v0.1.0 > SHA256SUMS
cd ..

# Step 4: Copy supporting files
cp configs/orvix.yaml dist/
cp scripts/install.sh dist/

# Step 5: Verify
ls -la dist/
# Expected: orvix-linux-amd64-v0.1.0, SHA256SUMS, orvix.yaml, install.sh
```
